Properties {
permission.readProperties - view only(have only view details action)
permission.editProperties - enables edit property button
permission.deleteProperties - enable the delete button

}

Plans {
permission.readPlans - view only action buttons will be removed from the table
permission.ceratePlan - enable create plan button
permission.editPlan - enable actions in the table
}


Addons {
permission.readAddon - view only action buttons will be removed from the table

permission.deleteAddon - enable delete "" "" ""

PropertyManagement {
permission.read - access property management page (view both property types and amenities)
permission.pmTCreate - create property type category and property type
permission.pmTEdit - edit property type
permission.pmTDelete - delete property type
permission.pmTStatus - toggle property type status (on/off)
permission.pmTManageFields - manage property type fields
permission.pmTOrder - manage property type order
permission.pmACreate - create amenity
permission.pmAEdit - edit amenity
permission.pmADelete - delete amenity
permission.pmAStatus - toggle amenity status (on/off)
permission.pmAOrder - manage amenity order
}


Filter Management {
